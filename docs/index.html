import litellm
from tokencost import CostTracker

# Create a tracker
tracker = CostTracker()
litellm.callbacks = [tracker]

# Make LLM calls as usual
response = litellm.completion(
    model="gpt-4",
    messages=[{"role": "user", "content": "Hello!"}]
)

# Check usage
print(f"Total: ${tracker.total_cost:.4f}")
print(f"Requests: {tracker.request_count}")
budget_control.py 
import litellm
from tokencost import CostTracker, BudgetExceededError

# Create a tracker with a $5 budget
def alert(tracker):
    print(f"Budget exceeded! Spent ${tracker.total_cost:.2f}")

tracker = CostTracker(
    budget=5.00,
    on_budget_exceeded=alert,
    raise_on_budget=True
)
litellm.callbacks = [tracker]

try:
    response = litellm.completion(
        model="gpt-4",
        messages=[{"role": "user", "content": "Hello!"}]
    )
except BudgetExceededError as e:
    print(f"Stopped at ${e.total_cost:.2f} (budget: ${e.budget:.2f})")
async_usage.py 
import asyncio
import litellm
from tokencost import CostTracker

async def main():
    tracker = CostTracker()
    litellm.callbacks = [tracker]

    # Works with async completions
    response = await litellm.acompletion(
        model="gpt-4",
        messages=[{"role": "user", "content": "Hello!"}]
    )

    print(f"Cost: ${tracker.total_cost:.6f}")

asyncio.run(main())
model_breakdown.py 
import litellm
from tokencost import CostTracker

tracker = CostTracker()
litellm.callbacks = [tracker]

# Make calls to different models
litellm.completion(model="gpt-4", messages=[...])
litellm.completion(model="gpt-3.5-turbo", messages=[...])
litellm.completion(model="claude-3-sonnet", messages=[...])

# Get cost breakdown by model
for model, cost in tracker.cost_by_model.items():
    print(f"{model}: ${cost:.6f}")

# Access full history
for entry in tracker.history:
    print(entry)
API Reference

Full Documentation

Everything you need to integrate cost tracking

CostTracker

Constructor Parameters

  • budget float | None

    Spending limit in USD. None means unlimited.

  • on_budget_exceeded Callable | None

    Callback function when budget is exceeded. Receives tracker instance.

  • raise_on_budget bool

    If True, raises BudgetExceededError when budget exceeded.

  • print_summary bool

    If True, prints cost summary when program exits.

Properties

Available on CostTracker instance

  • total_cost float

    Running total cost in USD.

  • request_count int

    Number of successful requests tracked.

  • history list[dict]

    All logged requests with model, tokens, cost, timestamp.

  • cost_by_model dict[str, float]

    Cost aggregated by model name.

  • budget_exceeded bool

    Whether the budget has been exceeded.

Automatic Exit Summary

When your program ends, you get a detailed cost report automatically. No extra code needed.

  • Total cost and request count
  • Budget status and remaining funds
  • Per-request breakdown with token counts
  • Disable with print_summary=False
